    <script>
        // Game state engine
        let state = {
            level: 1, xp: 0, maxXp: 100, points: 0, floor: 0,
            stats: { power: 0, defence: 0, speed: 0, mana: 0 },
            mana: 10, gold: 0, weapon: "Fists", weaponBonus: 0,
            shadows: { Common: 0, Uncommon: 0, Rare: 0 }
        };

        // Real-time 1 second = 1 XP clock tick
        setInterval(() => {
            state.xp += 1;
            if (state.xp >= state.maxXp) {
                state.xp -= state.maxXp;
                state.level += 1;
                state.points += 30;
                state.maxXp = Math.floor(state.maxXp * 1.5);
                logAction(`✨ SYSTEM UPDATE: Gained Level ${state.level}! +30 Stat Points Awarded.`);
                checkWeaponUnlock();
            }
            updateDOM();
        }, 1000);

        // Core Mathematical Calculators matching player rules
        function getMaxHp() { return Math.floor(100 * (2 ** (state.stats.defence / 10))); }
        function getTotalDps() { return Math.floor((15 + state.weaponBonus) * (2 ** (state.stats.power / 10))); }
        function getDodge() { return Math.min(90, (state.stats.speed / 10) * 10); }

        function addStat(type) {
            if (state.points > 0) { state.stats[type]++; state.points--; logAction(`✅ Allocated 1 point to ${type.toUpperCase()}.`); updateDOM(); }
            else { logAction("❌ You have no unassigned stat points available."); }
        }
        function buyShadow(type, cost) {
            if (state.mana >= cost) { state.mana -= cost; state.shadows[type]++; logAction(`🔮 Arise! A [${type}] Shadow Soldier joins your ranks.`); updateDOM(); }
            else { logAction("❌ Not enough Mana points to call forth shadows."); }
        }
        function mergeShadows(src, tgt, need) {
            if (state.shadows[src] >= need) { state.shadows[src] -= need; state.shadows[tgt]++; logAction(`🧬 Evolution Core: Fused ${need} ${src} into 1x ${tgt}!`); updateDOM(); }
            else { logAction(`❌ You need at least ${need} ${src} units to complete this merge transaction.`); }
        }
        function buyMana(cost, reward) {
            if (state.gold >= cost) { state.gold -= cost; state.mana += reward; logAction(`💧 Potion Consumed: +${reward} MP added to pool.`); updateDOM(); }
            else { logAction("❌ Insufficient Gold dropped from current dungeon clears."); }
        }

        function checkWeaponUnlock() {
            if (state.level >= 5 && state.weaponBonus < 15) { state.weapon = "Iron Dagger"; state.weaponBonus = 15; logAction("⚔️ Armory Unlock: Equipped Iron Dagger!"); }
            if (state.level >= 15 && state.weaponBonus < 45) { state.weapon = "Steel Sword"; state.weaponBonus = 45; logAction("⚔️ Armory Unlock: Equipped Steel Sword!"); }
        }

        function combatTrigger() {
            if (state.floor === 0) { state.floor = 1; logAction("🎯 Tutorial Cleared! Walking forward into Floor 1 Grid."); updateDOM(); return; }
            logAction(`⚔️ Striking Floor ${state.floor} Boss...`);
